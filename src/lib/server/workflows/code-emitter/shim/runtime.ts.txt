/**
 * Vendored runtime shim for emitted workflows.
 *
 * This file ships alongside the emitted workflow in the code_function's
 * supporting_files. Edit only if the workflow-builder source shim at
 * src/lib/server/workflows/code-emitter/shim/runtime.ts.txt is changed and
 * you need to re-emit every saved workflow.
 *
 * Transport: ctx.callActivity POSTs to FN_ROUTER_URL/execute. Inside the
 * code-runtime pod this defaults to the in-cluster function-router service
 * DNS; out-of-cluster runs set FN_ROUTER_URL explicitly.
 */

type Json = null | boolean | number | string | Json[] | { [key: string]: Json };

export interface WorkflowContextOptions {
	routerUrl?: string;
	routerToken?: string;
	trigger?: Record<string, unknown>;
	executionId?: string;
}

export class UnsupportedJqExpressionError extends Error {
	constructor(expression: string) {
		super(`Unsupported jq expression in emitted workflow: ${expression}`);
	}
}

export class WorkflowContext {
	readonly state: Record<string, unknown> = {};
	readonly #tasks = new Map<string, unknown>();
	readonly #routerUrl: string;
	readonly #routerToken?: string;
	readonly #trigger: Record<string, unknown>;
	readonly #executionId?: string;

	constructor(opts: WorkflowContextOptions = {}) {
		this.#routerUrl =
			opts.routerUrl ??
			(typeof process !== 'undefined' && process.env?.FN_ROUTER_URL) ??
			'http://function-router.workflow-builder.svc.cluster.local:8080';
		this.#routerToken =
			opts.routerToken ??
			(typeof process !== 'undefined' ? process.env?.FN_ROUTER_TOKEN : undefined);
		this.#trigger = opts.trigger ?? {};
		this.#executionId = opts.executionId;
	}

	async callActivity(slug: string, args: unknown): Promise<unknown> {
		const url = `${this.#routerUrl.replace(/\/$/, '')}/execute`;
		const headers: Record<string, string> = { 'content-type': 'application/json' };
		if (this.#routerToken) headers.authorization = `Bearer ${this.#routerToken}`;
		const response = await fetch(url, {
			method: 'POST',
			headers,
			body: JSON.stringify({
				function_slug: slug,
				input: args,
				execution_id: this.#executionId,
				node_name: slug,
				trigger_data: this.#trigger,
			}),
		});
		if (!response.ok) {
			const text = await response.text().catch(() => '');
			throw new Error(`function-router ${slug} failed: ${response.status} ${text}`);
		}
		const body = (await response.json()) as { data?: unknown; error?: string };
		if (body.error) throw new Error(`${slug}: ${body.error}`);
		this.#tasks.set(slug, body.data);
		return body.data;
	}

	async callUnsupportedTask(kind: string, raw: unknown): Promise<unknown> {
		console.warn(`[workflow-shim] Unsupported task kind "${kind}" executed as no-op.`, raw);
		return null;
	}

	setState(patch: Record<string, unknown>): void {
		Object.assign(this.state, patch);
	}

	jq(expression: string): unknown {
		return this.#evaluate(expression);
	}

	jqBool(expression: string): boolean {
		const value = this.#evaluate(expression);
		return Boolean(value);
	}

	jqArray(expression: string): unknown[] {
		const value = this.#evaluate(expression);
		if (Array.isArray(value)) return value;
		if (value == null) return [];
		throw new Error(
			`jq expression "${expression}" did not resolve to an array (got ${typeof value})`,
		);
	}

	errorMatches(err: unknown, filter: { type?: string; status?: number }): boolean {
		if (!err) return false;
		const e = err as { name?: string; code?: string; status?: number };
		if (filter.type && e.name !== filter.type && e.code !== filter.type) return false;
		if (filter.status !== undefined && e.status !== filter.status) return false;
		return true;
	}

	errorMatchesJq(err: unknown, expression: string): boolean {
		const snapshot = this.#withError(err);
		return Boolean(snapshot.evaluate(expression));
	}

	errorPayload(err: unknown): Record<string, unknown> {
		if (!err || typeof err !== 'object') return { message: String(err) };
		const e = err as { message?: string; name?: string; stack?: string };
		return { message: e.message ?? '', name: e.name ?? 'Error', stack: e.stack ?? '' };
	}

	async sleep(iso8601: string): Promise<void> {
		const ms = parseIso8601Duration(iso8601);
		if (ms > 0) await new Promise((resolve) => setTimeout(resolve, ms));
	}

	#withError(err: unknown): WorkflowContext {
		const clone = new WorkflowContext({
			routerUrl: this.#routerUrl,
			routerToken: this.#routerToken,
			trigger: this.#trigger,
			executionId: this.#executionId,
		});
		Object.assign(clone.state, this.state, { error: this.errorPayload(err) });
		return clone;
	}

	evaluate(expression: string): unknown {
		return this.#evaluate(expression);
	}

	#evaluate(expression: string): unknown {
		const trimmed = expression.trim();
		if (!trimmed) return undefined;
		// Literals
		if (trimmed === 'true') return true;
		if (trimmed === 'false') return false;
		if (trimmed === 'null') return null;
		if (/^-?\d+(\.\d+)?$/.test(trimmed)) return Number(trimmed);
		if (/^".*"$/.test(trimmed) || /^'.*'$/.test(trimmed)) return trimmed.slice(1, -1);

		// Comparisons / boolean composition
		const ops: Array<[RegExp, (l: unknown, r: unknown) => unknown]> = [
			[/\s+and\s+/, (l, r) => Boolean(l) && Boolean(r)],
			[/\s+or\s+/, (l, r) => Boolean(l) || Boolean(r)],
			[/\s*==\s*/, (l, r) => l === r],
			[/\s*!=\s*/, (l, r) => l !== r],
			[/\s*>=\s*/, (l, r) => (l as number) >= (r as number)],
			[/\s*<=\s*/, (l, r) => (l as number) <= (r as number)],
			[/\s*>\s*/, (l, r) => (l as number) > (r as number)],
			[/\s*<\s*/, (l, r) => (l as number) < (r as number)],
			[/\s*\+\s*/, (l, r) => (l as number) + (r as number)],
			[/\s*-\s*(?!>)/, (l, r) => (l as number) - (r as number)],
		];
		for (const [pattern, fn] of ops) {
			const parts = splitOnce(trimmed, pattern);
			if (parts) {
				return fn(this.#evaluate(parts[0]), this.#evaluate(parts[1]));
			}
		}

		// Path access
		return this.#resolvePath(trimmed);
	}

	#resolvePath(path: string): unknown {
		let scope: unknown;
		let rest = path;
		if (rest.startsWith('$.trigger')) {
			scope = this.#trigger;
			rest = rest.slice('$.trigger'.length);
		} else if (rest.startsWith('$task')) {
			scope = Object.fromEntries(
				[...this.#tasks].map(([key, value]) => [key, { output: value }]),
			);
			rest = rest.slice('$task'.length);
		} else if (rest.startsWith('$')) {
			scope = this.state;
			rest = rest.slice(1);
		} else {
			scope = { trigger: this.#trigger, ...this.state };
		}

		if (!rest || rest === '.') return scope;

		const steps = rest.match(/\.[a-zA-Z_][a-zA-Z0-9_]*|\[\d+\]|\[\*\]|\[\]/g) ?? [];
		let value: unknown = scope;
		for (const step of steps) {
			if (value == null) return undefined;
			if (step.startsWith('.')) {
				value = (value as Record<string, unknown>)[step.slice(1)];
			} else if (step === '[]' || step === '[*]') {
				if (!Array.isArray(value)) return undefined;
				return value;
			} else {
				const idx = Number(step.slice(1, -1));
				value = Array.isArray(value) ? value[idx] : undefined;
			}
		}
		return value;
	}
}

function splitOnce(
	input: string,
	pattern: RegExp,
): [string, string] | null {
	const match = input.match(pattern);
	if (!match || match.index === undefined) return null;
	return [
		input.slice(0, match.index).trim(),
		input.slice(match.index + match[0].length).trim(),
	];
}

function parseIso8601Duration(iso: string): number {
	const m = iso.match(/^PT(?:(\d+)H)?(?:(\d+)M)?(?:(\d+(?:\.\d+)?)S)?$/);
	if (!m) return 0;
	const hours = Number(m[1] ?? 0);
	const minutes = Number(m[2] ?? 0);
	const seconds = Number(m[3] ?? 0);
	return (hours * 3600 + minutes * 60 + seconds) * 1000;
}

export async function runWorkflow<T, R>(
	main: (ctx: WorkflowContext) => Promise<R>,
	trigger: T,
): Promise<R> {
	const ctx = new WorkflowContext({
		trigger: (trigger as Record<string, unknown>) ?? {},
	});
	return main(ctx);
}
